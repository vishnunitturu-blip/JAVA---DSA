import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {
        Scanner z=new Scanner(System.in);
        int n=z.nextInt();
        int []arr=new int[n];
        for(int i=0;i<n;i++){
            arr[i]=z.nextInt();
        }
        for(int i=1;i<n;i++){
        if(arr[i-1]>arr[i]){
            System.out.println("No");
            return;
            }
        }
        System.out.println("Yes");
    }
}
